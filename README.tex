\section{What this is}

Reproducibility artifact for the working paper
\textit{The Ledger of Meluhha --- Indus Valley Script as
Metrological Accounting Code} (Venugopal 2026).

Every claim in the paper is backed by data in the SQLite
databases. Every computation is reproducible with the F\#
scripts. No proprietary tools, no cloud services, no API keys.

Build instructions are in \texttt{CMakeLists.txt}. Read it.
It checks every dependency, tells you what is missing, and
gives the install command. Run:

\begin{verbatim}
cmake -S . -B build -DINDUS_VALLEY_DIR=../indus-valley
cmake --build build
\end{verbatim}

\section{Compile the paper (Overleaf)}

\begin{enumerate}[nosep]
\item Go to \url{https://www.overleaf.com} (free account).
\item Upload \texttt{ledger\_of\_meluhha.tex} +
      \texttt{citations.lua}.
\item Set compiler to \textbf{LuaLaTeX} (Menu $\to$ Compiler).
\item Recompile.
\end{enumerate}

\section{Files}

\begin{description}[style=nextline,leftmargin=2em]

\item[\texttt{ledger\_of\_meluhha.tex}]
  The paper. LuaLaTeX. 22 pages.

\item[\texttt{citations.lua}]
  Lua citation engine. First-use full citation, subsequent
  \textit{op.\ cit.}

\item[\texttt{IndusCodebookTypes.fsx}]
  Frozen typed F\# records for the metrological codebook:
  28~sign roles, 8~commodities, 5~routes, 11~weight tiers,
  5~merchant marks. Generated once by SqlHydra v4 from
  \texttt{indus\_codebook.db}. Zero runtime DB dependency.
  All decoder scripts \texttt{\#load} this file.

\item[\texttt{indus\_seed\_codebook.fsx}]
  Seeds \texttt{indus\_codebook.db}. Source of truth for
  the codebook schema. Raw ADO.NET (write path).

\item[\texttt{indus\_decoder.fsx}]
  Cargo-tag decoder. Uses frozen types, no arguments, no DB.
  \texttt{dotnet fsi indus\_decoder.fsx}

\item[\texttt{indus\_tn\_decoder.fsx}]
  Tamil Nadu cross-corpus decoder. Frozen codebook +
  corpus DB for CISI/concordance/parallels.
  \texttt{dotnet fsi indus\_tn\_decoder.fsx indus\_corpus.db}

\item[\texttt{indus\_ingest\_corpus.fsx}]
  Builds \texttt{indus\_corpus.db} from external sources.

\item[\texttt{indus\_lssc.fsx}]
  Transition entropy, LSSC scores, TSF Condition~2 check.

\item[\texttt{indus\_bernoulli.fsx}]
  Bernoulli independence argument. MathNet.Numerics.
  Analytic + Monte Carlo ($10^7$ trials) + Fisher's combined
  test + Bayesian update + sensitivity analysis.
  \texttt{dotnet fsi indus\_bernoulli.fsx}

\item[\texttt{ledger-of-meluhha.html}]
  Interactive dashboard. Single HTML file. Drop
  \texttt{indus\_corpus.db} onto the page.

\item[\texttt{spec/indus\_corpus.als}]
  Alloy 6 specification. 16~assertions, all UNSAT at scope~6.

\item[\texttt{spec/}]
  AI instruction contracts: F\# coding standard, TypeScript
  dashboard rules, LaTeX paper standard, CMake build rules,
  data contract.

\item[\texttt{hydra/}]
  SqlHydra v4 configuration and generated \texttt{.fs} types.
  Run once to regenerate if codebook schema changes.

\end{description}

\section{Query the results}

\begin{verbatim}
sqlite3 indus_codebook.db "SELECT * FROM sign_role WHERE role='commodity'"
sqlite3 indus_corpus.db "SELECT * FROM morphological_parallel"
sqlite3 indus_lssc.db "SELECT class, COUNT(*), ROUND(AVG(entropy),4)
                        FROM sign_entropy GROUP BY class"
\end{verbatim}
